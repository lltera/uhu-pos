  <h1>UHU Kimbap POS</h1>

  <div class="menu-list" id="menuList"></div>

  <div class="sales-log">

    <h2>Sales Log</h2>

    <table>

      <thead>

        <tr><th>Time</th><th>Item</th><th>Qty</th><th>Total (THB)</th></tr>

      </thead>

      <tbody id="logBody"></tbody>

    </table>

  </div>


  <script>

    const menu = [

      { name: "Original Kimbap", price: 69 },

      { name: "Keto Kimbap", price: 79 },

      { name: "Bibimbap", price: 89 },

      { name: "Kimchi Soup", price: 59 },

      { name: "Tteokbokki", price: 69 }

    ];


    const menuList = document.getElementById('menuList');

    const logBody = document.getElementById('logBody');


    function addLog(item, qty) {

      const now = new Date();

      const row = document.createElement('tr');

      row.innerHTML = `

        <td>${now.toLocaleTimeString()}</td>

        <td>${item.name}</td>

        <td>${qty}</td>

        <td>${item.price * qty}</td>

      `;

      logBody.appendChild(row);

    }


    menu.forEach(item => {

      const div = document.createElement('div');

      div.className = 'menu-item';

      div.innerHTML = `

        <span>${item.name} - ${item.price} THB</span>

        <div>

          <input type="number" min="1" value="1" id="qty-${item.name}" style="width:50px;">

          <button class="btn" onclick='addLog(${JSON.stringify(item)}, parseInt(document.getElementById("qty-${item.name}").value))'>Sell</button>

        </div>

      `;

      menuList.appendChild(div);

    });

  </script>
